# BarChart

柱状图组件

## 使用方法

```tsx
'use strict';

const React = require('react');
const ReactDom = require('react-dom');

import BarChart from 'yunke/charts/src/barChart';

export default class App extends React.Component {
  constructor(props) {
    super(props);
    this.state = {
      loading: true,
      data: [],
      opt: {
        keyLabel: 'label',
        keyValue: 'value',
      },
    };
  }
  componentDidMount() {
    //模拟异步加载数据
    setTimeout(() => {
      var data = [];
      for (let i = 0; i < 50; i++) {
        data.push({
          id: i + '',
          label: i,
          value: (100 + 50 * Math.random()).toFixed(0),
        });
      }
      this.setState({
        loading: false,
        data: data,
        total: data.length,
      });
    }, 1000);
  }
  onSelect(index, data) {
    console.log(typeof index);
    console.log(data);
  }
  render() {
    return (
      <BarChart
        data={this.state.data}
        total={this.state.total}
        opt={this.state.opt}
        onSelect={this.onSelect}
        loading={this.state.loading}
        barDesc={'单位:万元'}
      /> 
    );
  }
}
```
